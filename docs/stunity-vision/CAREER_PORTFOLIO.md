model Portfolio {
  id           String   @id @default(cuid())
  userId       String   @unique
  user         User     @relation(fields: [userId], references: [id])
  
  // Public URL
  username     String   @unique // stunity.com/@username
  customDomain String?  // Custom domain for premium users
  
  // Privacy
  isPublic     Boolean  @default(true)
  visibility   Visibility @default(PUBLIC)
  
  // Content
  headline     String?  // "Computer Science Student"
  summary      String?  // About me
  featuredProjects String[] // Array of project IDs
  
  // Analytics
  views        Int      @default(0)
  downloads    Int      @default(0)
  shares       Int      @default(0)
  
  // Settings
  theme        String   @default("default")
  layout       String   @default("standard")
  
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt
}

model Project {
  id           String   @id @default(cuid())
  userId       String
  user         User     @relation(fields: [userId], references: [id])
  
  title        String
  description  String
  thumbnail    String?
  
  skills       String[] // Technologies used
  category     String   // Web, Mobile, Data, etc.
  
  demoUrl      String?
  githubUrl    String?
  youtubeUrl   String?
  
  files        File[]   // Associated files
  
  grade        String?  // If graded
  courseId     String?  // If part of course
  
  isFeatured   Boolean  @default(false)
  isPublic     Boolean  @default(true)
  
  views        Int      @default(0)
  likes        Int      @default(0)
  
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt
}

model Skill {
  id           String   @id @default(cuid())
  userId       String
  user         User     @relation(fields: [userId], references: [id])
  
  name         String
  category     String   // Technical, Soft, Language
  proficiency  Int      // 0-100
  
  // Evidence
  courses      String[] // Course IDs
  projects     String[] // Project IDs
  certificates String[] // Certificate IDs
  
  verified     Boolean  @default(false)
  endorsements Endorsement[]
  
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt
}

model Endorsement {
  id           String   @id @default(cuid())
  skillId      String
  skill        Skill    @relation(fields: [skillId], references: [id])
  endorserId   String
  endorser     User     @relation(fields: [endorserId], references: [id])
  
  comment      String?
  
  createdAt    DateTime @default(now())
}

model PortfolioView {
  id           String   @id @default(cuid())
  portfolioId  String
  portfolio    Portfolio @relation(fields: [portfolioId], references: [id])
  
  viewerId     String?  // Null if anonymous
  viewer       User?    @relation(fields: [viewerId], references: [id])
  
  ipAddress    String?
  userAgent    String?
  referer      String?
  
  createdAt    DateTime @default(now())
}

model EmployerProfile {
  id           String   @id @default(cuid())
  userId       String   @unique
  user         User     @relation(fields: [userId], references: [id])
  
  companyName  String
  industry     String
  size         String   // Startup, SME, Enterprise
  website      String?
  
  verified     Boolean  @default(false)
  
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt
}
